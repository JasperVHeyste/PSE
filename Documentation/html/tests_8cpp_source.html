36bool FileCompare(const std::string leftFileName, const std::string rightFileName) { //van tictactoe14
37 ifstream leftFile, rightFile;
38 char leftRead, rightRead;
39 bool result;
40
41 //test if both files can be opened
42 leftFile.open(leftFileName.c_str());
43 if (!leftFile.is_open()) {
44 return false;
45 };
46 rightFile.open(rightFileName.c_str());
47 if (!rightFile.is_open()) {
48 leftFile.close();
49 return false;
50 };
51
52 result = true; // files exist and are open; assume equality unless a counterexamples shows up.
53 while (result && leftFile.good() && rightFile.good()) {
54 leftFile.get(leftRead);
55 rightFile.get(rightRead);
56 result = (leftRead == rightRead);
57 };
58 if (result) {
59 // last read was still equal; are we at the end of both files ?
60 result = (!leftFile.good()) && (!rightFile.good());
61 };
62
63 leftFile.close();
64 rightFile.close();
65 return result;
66}
